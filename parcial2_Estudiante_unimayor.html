Condicionales de verdad


from logic import *

# Definir los símbolos
rain = Symbol("rain")          # Llueve
bbc = Symbol("bbc")            # Estudiantes visitan BBC
unimayor = Symbol("unimayor")  # Estudiantes visitan Unimayor

# Conjunto de conocimientos
knowledge = And(
    Implication(Not(rain), bbc),           # Si no llueve, los estudiantes visitan BBC
    Or(bbc, unimayor),                     # Los estudiantes visitaron BBC o Unimayor
    Not(And(bbc, unimayor)),               # Pero no ambos
    unimayor                               # Los estudiantes visitaron Unimayor hoy
)

# Comprobar si el conocimiento implica que llueve
print(model_check(knowledge, rain))
